TABLE ARTICLE
	ID PRIMARY KEY AUTO
	OPEN_DATE DATETIME DEFAULT=NOW
	CLOSE_DATE DATETIME 
	DUE_DATE DATETIME
	IMG BLOB
	TITLE VARCHAR NOT NULL
	DESCRIPTION BLOB NOT NULL MIN_LENGTH(50)
	PROGRESS INT NOT NULL
	
	RELATED_ISSUES -> FOREIGN_KEY ISSUES
	
	STATUS_ID RELATIONSHIPS -> STATUS
	SUBTASK_ID RELATIONSHIPS -> SELF
	ASSIGNEE_ID RELATIONSHIP -> USER
	CATEGORY_ID RELATIONSHIP -> CATEGORY
	TAG_ID RELATIONSHIP -> TAG
	
	
TABLE USER
	ID PRIMAMRY_KEY
	USERNAME VARCHAR
	
	ARTICLE_ID FOREIGN_KEY -> ARTICLE
	
	
TABLE ISSUES
	ID PRIMARY KEY AUTO
	OPEN_DATE DATETIME DEFAULT=NOW
	CLOSE_DATE DATETIME
	DUE_DATE DATETIME
	TITLE VARCHAR NOT NULL
	DESCRIPTION BLOB NOT NULL MIN_LENGTH(50)
	STATUS VARCHAR NOT NULL
	CATEGORY VARCHAR NOT NULL
	PROGRESS INT NOT NULL
	
	ARTICLE_ID FOREIGN_KEY -> ARTICLE
	ASSIGNEE_ID FOREIGN_KEY -> USER
	

TABLE CATEGORY
	ID PRIMARY KEY AUTO
	TITLE VARCHAR NOT NULL
	DESCRIPTION BLOB NOT NULL MIN_LENGTH(50)
	
	ARTICLE_ID FOREIGN_KEY -> ARTICLE
	

TABLE TAG
	ID PRIMARY KEY AUTO
	TAG VARCHAR NOT NULL
	
	ARTICLE_ID FOREIGN_KEY -> ARTICLE
	
	
TABLE CONNECTION_TABLE
	ARTICLE_ID
	ASSIGNEE_ID
	

TABLE STATUS
	ID PRIMARY_KEY
